# Tipos de datos

```
.:Texto:. 
str | x = "Hello World"

.:Numerico:. 
int | x = 20
float | x = 20.5
complex | x =1j

.:Secuencial:.
list | x = ["apple", "banana", "cherry"]
tuple | x = ("apple", "banana", "cherry")
range | x = range(6)

.:Mapeo:.
dict | x = {"name" : "John", "age" : 36}

.:Asignación:.
set | x = {"apple", "banana", "cherry"}
frozenset | x = frozenset({"apple", "banana", "cherry"})

.:Booleano:.
bool | x = True

.:Binario:.
bytes | x = b"Hello"
bytearray | x = bytearray(5)
memoryview | x = memoryview(bytes(5))
```
